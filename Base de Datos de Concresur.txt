2. PERMISOS
Modelo simple, fiel al diseño de la UI:
 permiso      →  cada checkbox que aparece en la pantalla de Roles
 rol_permiso  →  tabla pivot: qué permisos tiene cada rol
 Flujo completo:
 1. Admin crea un rol  (ej: "vendedor")
 2. Admin tilda los checkboxes → se insertan filas en rol_permiso
 3. Admin crea un usuario y le asigna ese rol
 4. El sistema verifica acceso consultando rol_permiso por el rol del usuario

CREATE TABLE permiso (
    id      SMALLSERIAL     PRIMARY KEY,
    clave   VARCHAR(50)     NOT NULL UNIQUE,  -- usado en el código para verificar acceso
    nombre  VARCHAR(80)     NOT NULL          -- etiqueta que se muestra en los checkboxes
);

-- Tabla pivot: qué permisos tiene cada rol
-- Es la única tabla que se consulta para verificar acceso
CREATE TABLE rol_permiso (
    rol_id      SMALLINT    NOT NULL REFERENCES rol(id)     ON DELETE CASCADE,
    permiso_id  SMALLINT    NOT NULL REFERENCES permiso(id) ON DELETE CASCADE,
    PRIMARY KEY (rol_id, permiso_id)
);

3. CATÁLOGO DE PRODUCTOS E INVENTARIO

CREATE TABLE producto (
    id          SERIAL          PRIMARY KEY,
    nombre      VARCHAR(150)    NOT NULL,
    descripcion TEXT,
    activo      BOOLEAN         NOT NULL DEFAULT TRUE,
    creado_en   TIMESTAMPTZ     NOT NULL DEFAULT NOW()
);

CREATE TABLE variante_producto (
    id              SERIAL          PRIMARY KEY,
    producto_id     INTEGER         NOT NULL REFERENCES producto(id),
    nombre          VARCHAR(150)    NOT NULL,   -- ej: "20x40 cm", "Doble malla"
    unidad_medida   VARCHAR(30)     NOT NULL DEFAULT 'unidad',
    precio_venta    NUMERIC(12,2)   NOT NULL DEFAULT 0,
    stock_actual    INTEGER         NOT NULL DEFAULT 0,
    stock_minimo    INTEGER         NOT NULL DEFAULT 0,
    activo          BOOLEAN         NOT NULL DEFAULT TRUE,
    creado_en       TIMESTAMPTZ     NOT NULL DEFAULT NOW()
);

Kardex: historial inmutable de movimientos de inventario

CREATE TABLE movimiento_inventario (
    id                  BIGSERIAL       PRIMARY KEY,
    variante_id         INTEGER         NOT NULL REFERENCES variante_producto(id),
    tipo                VARCHAR(10)     NOT NULL,   -- 'ENTRADA' | 'SALIDA' | 'AJUSTE'
    cantidad            INTEGER         NOT NULL,
    stock_resultante    INTEGER         NOT NULL,   -- snapshot post-movimiento
    referencia_tipo     VARCHAR(30),                -- 'registro_produccion' | 'detalle_venta' | 'ajuste_manual'
    referencia_id       INTEGER,                    -- FK lógica al registro origen
    observacion         TEXT,
    creado_en           TIMESTAMPTZ     NOT NULL DEFAULT NOW()
);

4. COTIZACIONES
v1.2: sin fecha_expiracion ni estado 'Expirada'
Estados: 'Pendiente' | 'Confirmada' | 'Anulada'

CREATE TABLE cotizacion (
    id              SERIAL          PRIMARY KEY,
    numero          VARCHAR(20)     NOT NULL UNIQUE,    -- COT-0001
    cliente_id      INTEGER         NOT NULL REFERENCES cliente(id),
    importe_total   NUMERIC(14,2)   NOT NULL DEFAULT 0,
    estado          VARCHAR(15)     NOT NULL DEFAULT 'Pendiente',
    -- 'Pendiente' | 'Confirmada' | 'Anulada'
    creado_en       TIMESTAMPTZ     NOT NULL DEFAULT NOW(),
    confirmada_en   TIMESTAMPTZ,
    venta_id        INTEGER,        -- FK a venta, se establece al confirmar (FK declarada abajo)
    usuario_id      INTEGER         REFERENCES usuario(id)
);

CREATE TABLE detalle_cotizacion (
    id                  SERIAL          PRIMARY KEY,
    cotizacion_id       INTEGER         NOT NULL REFERENCES cotizacion(id) ON DELETE CASCADE,
    variante_id         INTEGER         NOT NULL REFERENCES variante_producto(id),
    cantidad            INTEGER         NOT NULL,
    precio_unitario     NUMERIC(12,2)   NOT NULL,
    subtotal            NUMERIC(14,2)   GENERATED ALWAYS AS (cantidad * precio_unitario) STORED
);

5. VENTAS

CREATE TABLE venta (
    id                  SERIAL          PRIMARY KEY,
    numero              VARCHAR(20)     NOT NULL UNIQUE,    -- VTA-0001
    cliente_id          INTEGER         NOT NULL REFERENCES cliente(id),
    cotizacion_id       INTEGER         REFERENCES cotizacion(id),  -- null si fue venta directa
    metodo_pago         VARCHAR(10)     NOT NULL,   -- 'Contado' | 'Credito'
    forma_pago          VARCHAR(10)     NOT NULL,   -- 'Efectivo' | 'QR'
    fecha_entrega       DATE,
    importe_total       NUMERIC(14,2)   NOT NULL DEFAULT 0,
    adelanto            NUMERIC(14,2)   NOT NULL DEFAULT 0,
    saldo_total         NUMERIC(14,2)   NOT NULL DEFAULT 0,
    estado_operativo    VARCHAR(20)     NOT NULL DEFAULT 'Confirmada',
    -- 'Confirmada' | 'En produccion' | 'Lista' | 'En camino' | 'Completada'
    estado_financiero   VARCHAR(10)     NOT NULL DEFAULT 'Pagado',
    -- 'Pendiente' | 'Pagado'
    usuario_id          INTEGER         REFERENCES usuario(id),
    creado_en           TIMESTAMPTZ     NOT NULL DEFAULT NOW(),
    completada_en       TIMESTAMPTZ
);

-- FK circular resuelta: cotizacion.venta_id → venta
ALTER TABLE cotizacion
    ADD CONSTRAINT fk_cotizacion_venta
    FOREIGN KEY (venta_id) REFERENCES venta(id);

CREATE TABLE detalle_venta (
    id                  SERIAL          PRIMARY KEY,
    venta_id            INTEGER         NOT NULL REFERENCES venta(id) ON DELETE CASCADE,
    variante_id         INTEGER         NOT NULL REFERENCES variante_producto(id),
    cantidad            INTEGER         NOT NULL,
    precio_unitario     NUMERIC(12,2)   NOT NULL,
    subtotal            NUMERIC(14,2)   GENERATED ALWAYS AS (cantidad * precio_unitario) STORED
);

CREATE TABLE abono_venta (
    id          SERIAL          PRIMARY KEY,
    venta_id    INTEGER         NOT NULL REFERENCES venta(id) ON DELETE CASCADE,
    monto       NUMERIC(14,2)   NOT NULL,
    forma_pago  VARCHAR(10)     NOT NULL,   -- 'Efectivo' | 'QR'
    usuario_id  INTEGER         REFERENCES usuario(id),
    creado_en   TIMESTAMPTZ     NOT NULL DEFAULT NOW()
);

6. PRODUCCIÓN

CREATE TABLE orden_produccion (
    id                      SERIAL          PRIMARY KEY,
    numero                  VARCHAR(20)     NOT NULL UNIQUE,    -- OP-0001
    variante_id             INTEGER         NOT NULL REFERENCES variante_producto(id),
    venta_id                INTEGER         REFERENCES venta(id),   -- null si es orden manual
    cantidad_requerida      INTEGER         NOT NULL,
    cantidad_producida      INTEGER         NOT NULL DEFAULT 0,
    estado                  VARCHAR(15)     NOT NULL DEFAULT 'Pendiente',
    -- 'Pendiente' | 'En proceso' | 'Completada'
    fecha_entrega_requerida DATE,
    creado_en               TIMESTAMPTZ     NOT NULL DEFAULT NOW(),
    completada_en           TIMESTAMPTZ,
    creado_por              INTEGER         REFERENCES usuario(id)
);

CREATE TABLE registro_produccion (
    id                  SERIAL          PRIMARY KEY,
    orden_id            INTEGER         NOT NULL REFERENCES orden_produccion(id),
    maestro_id          INTEGER         NOT NULL REFERENCES personal(id),
    cantidad_fabricada  INTEGER         NOT NULL,
    fecha               DATE            NOT NULL DEFAULT CURRENT_DATE,
    creado_en           TIMESTAMPTZ     NOT NULL DEFAULT NOW()
);

-- Relación N:M entre registro_produccion y personal auxiliar
CREATE TABLE registro_produccion_personal (
    registro_id     INTEGER     NOT NULL REFERENCES registro_produccion(id) ON DELETE CASCADE,
    personal_id     INTEGER     NOT NULL REFERENCES personal(id),
    PRIMARY KEY (registro_id, personal_id)
);

7. JORNAL

CREATE TABLE registro_jornal (
    id              SERIAL          PRIMARY KEY,
    personal_id     INTEGER         NOT NULL REFERENCES personal(id),
    fecha           DATE            NOT NULL DEFAULT CURRENT_DATE,
    hora_entrada    TIME            NOT NULL,
    hora_salida     TIME,
    observacion     TEXT,
    creado_en       TIMESTAMPTZ     NOT NULL DEFAULT NOW(),

    CONSTRAINT ck_jornal_horario CHECK (hora_salida IS NULL OR hora_salida > hora_entrada)
);

9. COMPRAS Y GASTOS

CREATE TABLE categoria_compra (
    id      SMALLSERIAL     PRIMARY KEY,
    nombre  VARCHAR(80)     NOT NULL UNIQUE
    -- 'Materia Prima' | 'Combustibles' | 'Lubricantes'
    -- 'Mantenimiento' | 'Gastos de Administración' | 'Otros'
);

CREATE TABLE compra (
    id                      SERIAL          PRIMARY KEY,
    numero                  VARCHAR(20)     NOT NULL UNIQUE,    -- CMP-0001
    categoria_id            SMALLINT        NOT NULL REFERENCES categoria_compra(id),
    proveedor_id            INTEGER         REFERENCES proveedor(id),   -- opcional
    metodo_pago             VARCHAR(10)     NOT NULL,   -- 'Contado' | 'Credito'
    forma_pago              VARCHAR(10)     NOT NULL,   -- 'Efectivo' | 'QR'
    importe_total           NUMERIC(14,2)   NOT NULL DEFAULT 0,
    adelanto                NUMERIC(14,2)   NOT NULL DEFAULT 0,
    saldo_total             NUMERIC(14,2)   NOT NULL DEFAULT 0,
    estado_financiero       VARCHAR(10)     NOT NULL DEFAULT 'Pagado',
    -- 'Pendiente' | 'Pagado'
    procesada_inventario    BOOLEAN         NOT NULL DEFAULT FALSE,  -- v2: entrada a inventario de insumos
    usuario_id              INTEGER         REFERENCES usuario(id),
    creado_en               TIMESTAMPTZ     NOT NULL DEFAULT NOW()
);

CREATE TABLE detalle_compra (
    id                  SERIAL          PRIMARY KEY,
    compra_id           INTEGER         NOT NULL REFERENCES compra(id) ON DELETE CASCADE,
    descripcion         VARCHAR(200)    NOT NULL,
    cantidad            NUMERIC(12,3),
    unidad_medida       VARCHAR(30),
    precio_unitario     NUMERIC(12,2)   NOT NULL,
    subtotal            NUMERIC(14,2)   GENERATED ALWAYS AS (
                            COALESCE(cantidad, 1) * precio_unitario
                        ) STORED,
    -- v2: control de materias primas (DEFAULT NULL en v1, sin interfaz activa)
    materia_prima_id    INTEGER,        -- FK a materia_prima (declarada abajo)
    cantidad_insumo     NUMERIC(12,3)   DEFAULT NULL
);

CREATE TABLE abono_compra (
    id          SERIAL          PRIMARY KEY,
    compra_id   INTEGER         NOT NULL REFERENCES compra(id) ON DELETE CASCADE,
    monto       NUMERIC(14,2)   NOT NULL,
    forma_pago  VARCHAR(10)     NOT NULL,   -- 'Efectivo' | 'QR'
    usuario_id  INTEGER         REFERENCES usuario(id),
    creado_en   TIMESTAMPTZ     NOT NULL DEFAULT NOW()
);

Fórmula de consumo de insumos por variante de producto
El campo 'activo' es el interruptor: FALSE en v1.2, TRUE activa el descuento en v2
CREATE TABLE formula_producto (
    id                  SERIAL          PRIMARY KEY,
    variante_id         INTEGER         NOT NULL REFERENCES variante_producto(id),
    materia_prima_id    INTEGER         NOT NULL REFERENCES materia_prima(id),
    cantidad_por_unidad NUMERIC(12,4)   NOT NULL,
    activo              BOOLEAN         NOT NULL DEFAULT FALSE,
    UNIQUE (variante_id, materia_prima_id)
);
