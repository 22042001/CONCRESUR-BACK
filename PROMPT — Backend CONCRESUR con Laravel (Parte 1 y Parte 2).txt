--- SERVICES ---
InventarioService:
  getStockActual(filtros):
    variantes con stock, precio, alerta_stock calculado
  getKardex(variante_id, filtros):
    historial movimientos DESC por creado_en
  ajustarStock(variante_id, cantidad, observacion):
    DB::transaction() + lockForUpdate() sobre variante
    stock_actual += cantidad (puede ser negativo)
    insertar movimiento_inventario tipo 'AJUSTE'
    stock_resultante = nuevo stock_actual

CotizacionService:
  crear(data, usuario_id):
    DB::transaction():
      generar numero COT-XXXX con lockForUpdate()
      crear cotizacion estado 'Pendiente'
      insertar detalle_cotizacion
      calcular y guardar importe_total
  confirmarComoVenta(id, data, usuario_id):
    Validar estado == 'Pendiente'
    DB::transaction():
      crear Venta via VentaService::crear()
      cotizacion.estado = 'Confirmada'
      cotizacion.confirmada_en = NOW()
      cotizacion.venta_id = nueva_venta.id
  anular(id):
    Validar estado == 'Pendiente'
    cotizacion.estado = 'Anulada'

VentaService:
  crear(data, usuario_id):
    DB::transaction():
      generar numero VTA-XXXX con lockForUpdate()
      Calcular estado_financiero inicial:
        Contado → estado_financiero='Pagado',
          adelanto=importe_total, saldo_total=0
        Credito → estado_financiero='Pendiente',
          saldo_total=importe_total-adelanto
      Crear venta
      Por cada item:
        leer variante con lockForUpdate()
        Si stock_actual >= cantidad:
          stock_actual -= cantidad
          insertar movimiento 'SALIDA'
            ref 'detalle_venta'
          insertar detalle_venta
        Si stock_actual < cantidad:
          venta.estado_operativo = 'En produccion'
          PENDIENTE ITER 3: llamar
            OrdenProduccionService::crearAutomatica()
          insertar detalle_venta sin descontar stock
      Actualizar importe_total = suma de subtotales
