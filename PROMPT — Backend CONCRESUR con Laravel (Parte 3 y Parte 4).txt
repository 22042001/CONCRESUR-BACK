--- MIGRACIONES (en este orden exacto) ---
1. create_categoria_compras_table
     id SMALLSERIAL PK
     nombre VARCHAR(80) NOT NULL UNIQUE
2. create_proveedors_table
     id, nombre VARCHAR(150), telefono VARCHAR(20)
     observaciones TEXT, activo, creado_en
3. create_compras_table
     id, numero VARCHAR(20) NOT NULL UNIQUE
     categoria_id SMALLINT FK categoria_compra
     proveedor_id FK proveedor NULL
     metodo_pago VARCHAR(10), forma_pago VARCHAR(10)
     importe_total NUMERIC(14,2) DEFAULT 0
     adelanto NUMERIC(14,2) DEFAULT 0
     saldo_total NUMERIC(14,2) DEFAULT 0
     estado_financiero VARCHAR(10) DEFAULT 'Pagado'
     procesada_inventario BOOLEAN DEFAULT FALSE
     usuario_id FK NULL, creado_en
4. create_detalle_compras_table
     id, compra_id FK ON DELETE CASCADE
     descripcion VARCHAR(200)
     cantidad NUMERIC(12,3) NULL
     unidad_medida VARCHAR(30) NULL
     precio_unitario NUMERIC(12,2)
     subtotal GENERATED ALWAYS AS
       (COALESCE(cantidad,1) * precio_unitario) STORED
     materia_prima_id INTEGER NULL (sin FK activa en v1)
     cantidad_insumo NUMERIC(12,3) NULL
5. create_abono_compras_table
     id, compra_id FK ON DELETE CASCADE
     monto NUMERIC(14,2), forma_pago VARCHAR(10)
     usuario_id FK NULL, creado_en
6. create_orden_produccions_table
     id, numero VARCHAR(20) NOT NULL UNIQUE
     variante_id FK variante_producto
     venta_id FK venta NULL
     cantidad_requerida INTEGER
     cantidad_producida INTEGER DEFAULT 0
     estado VARCHAR(15) DEFAULT 'Pendiente'
       valores: 'Pendiente'|'En proceso'|'Completada'
     fecha_entrega_requerida DATE NULL
     creado_en, completada_en TIMESTAMPTZ NULL
     creado_por FK usuario NULL
7. create_registro_produccions_table
     id, orden_id FK orden_produccion
     maestro_id FK personal
     cantidad_fabricada INTEGER
     fecha DATE DEFAULT CURRENT_DATE, creado_en
8. create_registro_produccion_personals_table
     PK compuesta (registro_id, personal_id)
     registro_id FK ON DELETE CASCADE
     personal_id FK personal
9. create_registro_jornals_table
     id, personal_id FK personal
     fecha DATE DEFAULT CURRENT_DATE
     hora_entrada TIME NOT NULL
     hora_salida TIME NULL
     observacion TEXT NULL, creado_en
     CONSTRAINT ck_jornal_horario:
       hora_salida IS NULL OR hora_salida > hora_entrada

--- SERVICES ---
CompraService:
  crear(data, usuario_id):
    DB::transaction():
      generar numero CMP-XXXX con lockForUpdate()
      Contado → estado_financiero='Pagado', saldo=0
      Credito → estado_financiero='Pendiente',
        saldo=importe_total-adelanto
      crear compra, insertar detalle_compra
      actualizar importe_total = suma subtotales
  getInformeMensual(mes, anio, categoria_id):
    compras filtradas con detalle y suma total

OrdenProduccionService:
  crearAutomatica(venta_id, variante_id, cantidad,
                  usuario_id, fecha_entrega):
    DB::transaction():
      verificar que no exista orden activa para
      la misma venta+variante (evitar duplicados)
      generar numero OP-XXXX con lockForUpdate()
      crear orden con venta_id
  crearManual(data, usuario_id):
    DB::transaction():
      generar numero OP-XXXX
      crear orden sin venta_id (null)
  getActivas():
    estado Pendiente o En proceso

RegistroProduccionService:
  guardar(data, usuario_id):
    Validar orden no está Completada
    DB::transaction():
      crear RegistroProduccion
      si personal_auxiliar: insertar en tabla pivot
      RegistroProduccionObserver se dispara solo
      OrdenProduccionObserver se dispara si la orden
        cambia a Completada
    retornar RegistroProduccionResource

JornalService:
  registrar(data):
    crear registro_jornal
    (constraint de BD garantiza hora_salida > hora_entrada)
  getHistorial(personal_id, filtros):
    filtro por fecha, ordenado DESC

--- SERVICE ---
LogisticaService:
  getKanban():
    Retorna pedidos agrupados:
    {
      "en_espera": [...],  ordenados ASC fecha_en_espera
      "en_camino": [...],  ordenados ASC fecha_en_camino
      "entregado":  [...]  últimos 7 días, DESC
    }
    Cada item: PedidoLogisticoResource completo
  moverEstado(id, nuevo_estado, observaciones,
              usuario_id):
    Validar transiciones permitidas:
      'En espera' → 'En camino'  (válida)
      'En camino' → 'Entregado'  (válida)
      cualquier otra → 422
    DB::transaction():
      pedido.estado = nuevo_estado
      pedido.observaciones = observaciones (si viene)
      PedidoLogisticoObserver actualiza fechas
        y venta automáticamente
